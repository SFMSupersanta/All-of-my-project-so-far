Question 2 - Walkthrough - (10 marks):

What is the exact output of the following program?  Show your rough work to avoid deductions. 
 #include <stdio.h>

 int card(char r, char s) {
     int a = 0;
     if (r == 'A' || r == 'a')
         a = 13;
     else if (r == 'K' || r == 'k')
         a = 12;
     else if (r == 'Q' || r == 'q')
         a = 11;
     else if (r == 'J' || r == 'j')
         a = 10;
     else if (r == '0')
         a = 9;
     else if (r > '1' && r <= '9')
         a = (r - '0') - 1; /* Note that 5 = '5' - '0' */ 
     printf("%c,%c:%2d,", r, s, a);
     if (a != 0) {
         switch (s) {
         case 'S':
         case 's':
             a = a + 39;
             break;
         case 'H':
         case 'h':
             a = a + 26;
             break;
         case 'D':
         case 'd':
             a = a + 13;
         case 'C':
         case 'c':
             break;
         default :
             a = 0;
         }
     }
     printf("%2d: ", a);
     return a;
 }

 int main(void) {
     printf("%d\n", card('A', 'D'));
     printf("%d\n", card('9', 'h'));
     printf("%d\n", card('1', 'C'));
     printf("%d\n", card('5', 'W'));
     return 0;
 }

Question 4 - Walkthrough - (10 marks):

What is the exact output of the following program?  Show your rough work to avoid deductions. 
 #include <stdio.h>
 #define D 234
 #define R 8

 int getX(int d, int r) {
     int x = 1;

     while (d != 0) {
         x = x * 10;
         d = d / 10;
     }
     printf("%d ", x);
     do {
         x = x / 10;
         r = r / 10;
     } while (r != 0);
     printf("%d ", x);

     return x / 10;
 }

 void getpq(int d, int x, int r, int *p, int *q) {

     *q = 0;
     if (x != 0) {
         *p = d / x;
         printf("%d %d %d\n", d, *p, x);
         while (x != 0) {
             *q = 10 * *q + *p / r;
             *p = 10 * (*p % r) + (d * 10 / x) % 10;
             printf("%d %d %d\n", *q, *p, x);
             x = x / 10;
         }
         *p = *p / 10;
     } else
         *p = 0;
 }

 int main (void) {
     int d = D, x, q, p;

     x = getX(d, R);
     getpq(d, x, R, &p, &q);
     printf("%d __ %d = %d\n", d, R, q);
     printf("%d __ %d = %d\n", d, R, p);

     return 0;
 }
