 void display(int a[], int no) {
     int i;
     for (i = 0; i < no; i++)
         printf("%03d ", a[i]);
     putchar('\n');

     while (inc < 0)
         inc += 60;
     *a = (*a + inc) % 60;
     return inc;

     int match = 1;
     while(no != 0 && match == 1) {
         --no;
         if (a[no] != b[no])
             match = 0;
     }
     if (match == 1)
         printf("Unlocked!\n");
     else
         printf("Locked\n");
