 void bubble(char a[][MAX_LETTERS+1], int size) { 
     int i, j;
     char temp[MAX_LETTERS+1];

     for (i = size - 1; i > 0; i--) {
         for (j = 0; j < i; j++) {
             if (strcmp(a[j],a[j+1]) > 0) {
                 strcpy(temp, a[j]);
                 strcpy(a[j], a[j+1]);
                 strcpy(a[j+1], temp);
             }
         }
     }
