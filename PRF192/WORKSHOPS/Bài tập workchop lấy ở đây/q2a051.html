What is the exact output of the following program? 

 #include <stdio.h>
 #define INITIAL {'U','N','I','X'}
 #define WORDSIZE 2
 #define WORDS 2

 void convert(char [], int, int);

 int main() {
     int i;
     char word[WORDSIZE * WORDS] = INITIAL;

     printf("Start with  ");
     for (i = 0; i < WORDSIZE * WORDS; i++)
         printf("%c", word[i]);
     printf("\n");
     convert(word, WORDSIZE, WORDS);
     printf("This is the ");
     for (i = 0; i < WORDSIZE * WORDS; i++)
         printf("%c", word[i]);
     printf(" problem\n");
     convert(word, WORDSIZE, WORDS);
     printf("Finish with ");
     for (i = 0; i < WORDSIZE * WORDS; i++)
         printf("%c", word[i]);
     printf("\n");

     return 0;
 }

 void convert(char word[], int wordSize, int nWords) {
     int i, j, k, m, n;
     char temp;

     k = wordSize / 2;
     for (i = 0; i < wordSize * nWords; i += wordSize) { 
         n = i + wordSize - 1;
         m = i;
         for (j = 0; j < k; j++) {
             temp = word[m];
             word[m++] = word[n];
             word[n--] = temp;
         }
     }
 }
