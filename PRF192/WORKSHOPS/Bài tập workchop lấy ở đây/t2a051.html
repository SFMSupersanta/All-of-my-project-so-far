 #include <stdio.h>
 #define N 3
 int e(int, int);

 int main( ) {
     int i, h;
     int a[] = {35, 24, 26};
     int b[] = {14, 18, 13};
     int c[] = {00, 00, 00};

     for (i = 0; i < N; i++)
         c[i] = e(a[i], b[i]);

     printf("i a[i] b[i] c[i]\n");
     for (i = 0; i < N; i++)
         printf("%1d%5d%5d%5d\n", i, a[i], b[i], c[i]); 

     return 0;
 }

 int e(int n, int m) {
     int r;

     do {
         r = m % n;
         if (r != 0) {
             m = n;
             n = r;
         }
     } while (r != 0);

     return n;
 }

Question 4 - Walkthrough - (10 marks):

What is the exact output of the following program?  Show your rough work to avoid deductions. 

 #include <stdio.h>
 #include <string.h>

 int at(char s[]) {
     int k;

     k = 0;
     while (s[k] != '@' && s[k] != '\0')
         k++;
     if (s[k] == '\0')
         k = -1;
     return k;
 }

 int dot(char s[], int k) {
     int n;

     n = 0;
     while (s[k] != '\0') {
         if (s[k] == '.')
             n++;
         k++;
     }
     return n;
 }

 void janet(char s[], int k, int d) {
     int t, m, n, p;

     n = strlen(s) - 1;
     do {
         p = n;
         do {
             t = s[k];
             for (m = k; m < n; m++)
                 s[m] = s[m+1];
             s[n--] = t;
         } while (t != '.');
         for (m = n + 2; m < p; m++) {
             t = s[m];
             s[m] = s[p];
             s[p] = t;
             p--;
         }
         d--;
     } while (d > 0);
 }

 int main( ) {
     int i, k, n;
     char s[2][14] = {"btp@cs.ca", "cs.ca"}; 

     for (i = 0; i < 2; i++) {
         printf("%s\n", s[i]);
         k = at(s[i]);
         if (k >= 0) {
             n = dot(s[i], k+1);
             janet(s[i], k+1, n);
         }
         printf("%s\n", s[i]);
     }
     return 0;
 }
