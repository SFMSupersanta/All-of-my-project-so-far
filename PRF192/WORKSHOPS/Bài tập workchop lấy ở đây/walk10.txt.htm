/* w10.c */ #include #define MAX 5 void fill(double x[], int sz); int find(double num, double x[], int sz); void show(double x[], int sz); main() { double nums[MAX]; int n; fill(nums, MAX); n = find(4.5, nums, MAX); show(nums, n); } void fill(double x[], int sz) { int i; for (i = 0; i < sz; i++) x[i] = 1.5 * (i + 1); } int find(double num, double x[], int sz) { int i = 0; double diff; do { diff = x[i] - num; printf("%.2lf\n", diff); i++; } while (i < sz && (diff < -0.5 || diff > 0.5)); return i; } void show(double x[], int sz) { int i; for (i = sz - 1; i >= 0; i--) printf("%.2lf ", x[i]); printf("\n"); }
