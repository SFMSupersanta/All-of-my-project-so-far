In-Class Practice #15 - String Input


Consider the following programs 


 int main() {
     char s[10] = "?";

     scanf("%s", s);

     return 0;
 }


 int main() {
     char s[10] = "?";

     scanf("%[^\n]", s);

     return 0;
 }


 int main() {
     char s[10] = "?";

     gets(s);

     return 0;
 }


 void getstr(char [], int);

 int main() {
     char s[10] = "?";

     getstr(s, 9);

     return 0;
 }


 void getstr(char s[], int max) {
     int i, c;

     i = 0;
     while((c = getchar()) != '\n' && c != EOF) 
         if (i < max)
             s[i++] = (char) c;
     s[i] = '\0';
 }
