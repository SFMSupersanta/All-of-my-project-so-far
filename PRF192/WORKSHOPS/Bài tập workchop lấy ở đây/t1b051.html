Question 2 - Walkthrough - (10 marks):

What is the exact output of the following program?  Show your rough work to avoid deductions. 

 #include <stdio.h>
 #define N 2
 int i2h(int);
 void showh(int);

 int main( ) {
     int i, h;

     i = 63;
     h = i2h(i);
     showh(h);
     i = 129;
     h = i2h(i);
     showh(h);

     return 0;
 }

 int i2h(int b) {
     int h = 0, i, j = 0, k, d = 1;

     for (i = 1; i < 4*N; i++)
         d = 2 * d;
     i = 0;
     printf("%04d is ", b);
     while (i < 4*N) {
         k = b / d;
         j = 2 * j + k;
         printf("%d", k);
         if (i % 4 == 3) {
             h = 16 * h + j;
             j = 0;
         }
         b = b % d;
         d = d / 2;
         i++;
     }
     return h;
 }

 void showh(int h) {
     int i, d = 1;

     for (i = 1; i < N; i++)
         d = 16 * d;
     i = 0;
     printf(", which is ");
     while (i < N) {
         printf("%x", h / d); /* hex representation */ 
         h = h % d;
         d = d / 16;
         i++;
     }
     putchar('\n');
 }


Question 3 - Word Problem - (10 marks):

Design and code a program that solves an ordinary differential equation using Euler's algorithm.  The algorithm takes the form 

 ynext = yprev + f(tprev, yprev) * step

where

  • f(tprev, yprev) is an expression for the derivative of y with respect to t evaluated at point (tprev,yprev) and
  • step is the size of the step between (tnext,ynext) and (tprev,yprev), so that  

     tnext = tprev + step

Your program
  • prompts for and accepts the initial value of y, the step size and the number of steps
  • includes input validation,
  • calls a function to determine the value of the derivative of y with respect to t at a point (t,y); and
  • displays the values of t, y, and f(t,y) at each step. 
For your solution, you may assume that the expression for the derivative of y with respect to t at point (t,y) is given by 

 f(y, t) = 1 - t + 4y

For this equation, a step size of 0.1 and 10 steps, your program displays results that look something like 

 Initial Value of y : 1.0
 Step Size          : 0.1
 Number of Steps    : 10

 Step      t     y  f(t,y)
 Initial 0.0   1.0     5.0
    1    0.1   1.5     6.9
    2    0.2   2.2     9.6
    3    0.3   3.1    13.3
    4    0.4   4.5    18.5
    5    0.5   6.3    25.8
    6    0.6   8.9    36.0
    7    0.7  12.5    50.3
    8    0.8  17.5    70.3
    9    0.9  24.6    98.4
   10    1.0  34.4   137.6

Design your program for high cohesion and low coupling and so that a change to the formula for the derivative of y with respect to t would involve a minimal amount of recoding.

 #define E 3

         m = m * 2;

         p = m / 2;

             t = t * 2.0;

             b = 2 * b;
         t = t / 2.0;
     }
     for (i = E; i >= 1; i--) {
         b = 2 * b + p % 2;
         p = p / 2;
     }
     return 2 * b + s;
 }
